    width_multiple: 0.50  # layer channel multiple 广度缩放系数，影响输出通道数（args中第一个元素）
    anchors: # 锚框（锚定框）：三个特征图，每个特征图上有三组锚框

      # YOLOv5 v6.0 backbone

        [[-1, 1, Conv, [64, 6, 2, 2]],  # 0-P1/2 网络编号-卷积编号/原始图像缩放倍数
